Database setup SQL

Run this once in your Postgres instance to create the three evaluation tables.

-- Create schema CREATE SCHEMA IF NOT EXISTS evaluation; -- 1) Gold test cases CREATE TABLE IF NOT EXISTS evaluation.auto_test_cases ( id BIGSERIAL PRIMARY KEY, question TEXT NOT NULL, gold_answer TEXT, intent VARCHAR(50), category VARCHAR(80), difficulty SMALLINT DEFAULT 1, is_active BOOLEAN NOT NULL DEFAULT TRUE, created_at TIMESTAMPTZ NOT NULL DEFAULT now(), notes TEXT ); -- 2) Per-case scores CREATE TABLE IF NOT EXISTS evaluation.auto_results ( id BIGSERIAL PRIMARY KEY, run_id BIGINT, test_case_id BIGINT REFERENCES evaluation.auto_test_cases(id), agent_answer TEXT, score_accuracy NUMERIC(3,1), score_helpfulness NUMERIC(3,1), score_safety NUMERIC(3,1), score_hallucination NUMERIC(3,1), score_overall NUMERIC(3,1), judge_rationale TEXT, judge_model VARCHAR(80), response_ms INTEGER, created_at TIMESTAMPTZ NOT NULL DEFAULT now() ); -- 3) Run aggregates CREATE TABLE IF NOT EXISTS evaluation.auto_runs ( id BIGSERIAL PRIMARY KEY, run_label VARCHAR(120), agent_version VARCHAR(80), total_cases INTEGER, avg_accuracy NUMERIC(3,1), avg_helpfulness NUMERIC(3,1), avg_safety NUMERIC(3,1), avg_hallucination NUMERIC(3,1), avg_overall NUMERIC(3,1), pass_rate NUMERIC(5,2), created_at TIMESTAMPTZ NOT NULL DEFAULT now() );
